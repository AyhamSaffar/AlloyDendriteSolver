// install brew if not on system

brew install enzyme

// generate intermediate output (LLVM)

clang++ test.cpp -S -emit-llvm -o input.ll -O2 -fno-vectorize -fno-slp-vectorize -fno-unroll-loops

// get enzyme install path

brew info enzyme

PATH = /home/linuxbrew/.linuxbrew/Cellar/enzyme/0.0.249_1/lib/LLVMEnzyme-22.so

// auto differentiate LLVM intermediate input into LLVM output

opt input.ll --load-pass-plugin=PATH -passes=enzyme -o output.ll -S

// optimise the output LLVM

opt output.ll -O2 -o output_opt.ll -S

clang++ output_opt.ll -o main
